\section*{Terminology}
\addcontentsline{toc}{section}{Terminology}

\begin{tabular}{p{3.8cm}|p{10cm}}
	\defterm{RFID} & Radio Frequency Identification\\
	\defterm{RFID Reader} & Initiates communication and interrogates tags\\
	\defterm{RFID Tag} & Responds to inquiries from readers\\
	\defterm{Passive RFID} & An RFID tag without an internal power source such as a battery. All energy needed to drive the circuitry must be harvested from its surroundings.\\
	\defterm{Inductive RFID} & RFID technology using magnetic fields for transferring data and powering the tags, as opposed to radiative RFID which uses electromagnetic fields.\\
	\defterm{Soft Tag} & An RFID tag programmed in software instead of using a fixed hardware state machine\\
\end{tabular}
